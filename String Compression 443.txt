from typing import List

class Solution:
    def compress(self, chars: List[str]) -> int:
        n = len(chars)
        i = 0
        result = []
        
        while i < n:
            count = 1
            while i + 1 < n and chars[i] == chars[i + 1]:
                count += 1
                i += 1
            
            result.append(chars[i])
            
            if count > 1:
                for c in str(count):
                    result.append(c)
            
            i += 1
        
        for j in range(len(result)):
            chars[j] = result[j]
        
        return len(result)
